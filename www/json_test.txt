    type: 'GET', 
    url: 'https://skynetweather.com/html/weather34/jsondata/realtimetest.json', 
    data: { get_param: 'value' }, 
    dataType: 'json',
    success: function (data) { 
        $.each(data, function(index, element) {
            $('body').append($('<div>', {
                text: element.value
            }));
        });
    }
});
